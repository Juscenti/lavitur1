<div class="toast-container" id="toasts"></div>

<script>
  let pendingDeleteName = '';

  function setMode(mode) {
    const tableEl = document.getElementById('table-mode');
    const editEl = document.getElementById('editorial-mode');
    const btnTable = document.getElementById('btn-table-mode');
    const btnEd = document.getElementById('btn-editorial-mode');
    if (mode === 'table') {
      tableEl.style.display = 'block';
      editEl.style.display = 'none';
      btnTable.classList.add('active');
      btnEd.classList.remove('active');
    } else {
      tableEl.style.display = 'none';
      editEl.style.display = 'block';
      btnEd.classList.add('active');
      btnTable.classList.remove('active');
    }
  }

  function toggleAll(master) {
    document.querySelectorAll('.row-check').forEach(cb => cb.checked = master.checked);
    updateBulkBar();
  }

  function updateBulkBar() {
    const checked = document.querySelectorAll('.row-check:checked').length;
    const bar = document.getElementById('bulk-bar');
    bar.classList.toggle('visible', checked > 0);
    document.getElementById('bulk-count').textContent = `${checked} selected`;
    document.querySelectorAll('tbody tr').forEach(tr => {
      tr.classList.toggle('selected', tr.querySelector('.row-check').checked);
    });
  }

  function filterTable() {
    // placeholder — would filter rows
    showToast('Filter applied', 'success');
  }

  function clearFilters() {
    document.querySelectorAll('.filter-select').forEach(s => s.value = 'all');
    document.querySelector('.search-input').value = '';
    showToast('Filters cleared', 'success');
  }

  function toggleCat(el) {
    el.classList.toggle('active');
  }

  function toggleSubCat(el) {
    el.classList.toggle('active');
  }

  function openModal(name = '') {
    pendingDeleteName = name;
    const body = document.getElementById('modal-body');
    body.textContent = name
      ? `Are you sure you want to delete "${name}"? This action cannot be undone.`
      : 'Are you sure you want to delete the selected products? This action cannot be undone.';
    document.getElementById('modal').classList.add('open');
  }

  function closeModal() {
    document.getElementById('modal').classList.remove('open');
  }

  function confirmDelete() {
    closeModal();
    showToast(pendingDeleteName ? `"${pendingDeleteName}" deleted` : 'Products deleted', 'error');
  }

  document.getElementById('modal').addEventListener('click', function(e) {
    if (e.target === this) closeModal();
  });

  function showToast(msg, type = 'success') {
    const container = document.getElementById('toasts');
    const t = document.createElement('div');
    t.className = `toast ${type}`;
    const icon = type === 'success'
      ? `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5"><polyline points="20 6 9 17 4 12"/></svg>`
      : `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="3 6 5 6 21 6"/><path d="M19 6l-1 14a2 2 0 0 1-2 2H8a2 2 0 0 1-2-2L5 6"/></svg>`;
    t.innerHTML = icon + msg;
    container.appendChild(t);
    setTimeout(() => { t.style.opacity = '0'; t.style.transition = 'opacity 0.3s'; setTimeout(() => t.remove(), 300); }, 2500);
  }
</script>
</body>
